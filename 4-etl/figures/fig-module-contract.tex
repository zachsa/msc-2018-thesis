(function () {
    // Execution context ('this') is a config object
    function invoke(obj) {
        return new Promise((resolve, reject) => {
            setImmediate(() => {
                try { resolve(obj); }
                catch (error) { reject(error); };
            });
        });
    };
    return new Promise(async(resolve, reject) => {
        setImmediate(() => {
            try { resolve({ invoke: invoke }); }
            catch (error) { reject(error); };
        });
    });
})();
